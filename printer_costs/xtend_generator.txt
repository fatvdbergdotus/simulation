    def CharSequence compile(Model model) '''

    x = np.sort(data)
    y = np.arange(1, len(x) + 1) / len(x)
    return x, y

def plot_ecdf(lower_bounds, upper_bounds):

    x1, y1 = ecdf(lower_bounds)
    x2, y2 = ecdf(upper_bounds)

    plt.figure(figsize=(10,6))
    plt.plot(x1, y1, label="Lower Bound")
    plt.plot(x2, y2, label="Upper Bound")

    plt.xlabel("Cost")
    plt.ylabel("Probability")
    plt.title("Empirical CDF")
    plt.legend()
    plt.grid(True)
    plt.show()

    def dispatch CharSequence generateDistribution(
        NormalDistribution d
    ) '''
while True:
    «d.variable.name» = random.gauss(
        «d.mean»,
        «d.stddev»
    )
    if «d.lower» <= «d.variable.name» <= «d.upper»:
        break
'''

    def dispatch CharSequence generateDistribution(
        UniformDistribution d
    ) '''
«d.variable.name» = random.uniform(
    «d.lower»,
    «d.upper»
)
'''

    def dispatch CharSequence generateDistribution(
        TriangularDistribution d
    ) '''
«d.variable.name» = random.triangular(
    «d.low»,
    «d.high»,
    «d.mode»
)
'''

    def dispatch CharSequence generateDistribution(
        ChoiceDistribution d
    ) '''
«d.variable.name» = random.choice(
    [«FOR v : d.values SEPARATOR ', '»«v»«ENDFOR»]
)
'''

    def String compileMin(
        Expression expr,
        Model model
    ) {

        var result = expr.toPython

        for (u : model.uncertainVariables) {
            result = result.replace(
                u.name,
                u.name + "_min"
            )
        }

        result
    }

    def String compileMax(
        Expression expr,
        Model model
    ) {

        var result = expr.toPython

        for (u : model.uncertainVariables) {
            result = result.replace(
                u.name,
                u.name + "_max"
            )
        }

        result
    }

    def dispatch String toPython(NumberLiteral e) {
        e.value.toString
    }

    def dispatch String toPython(VariableRef e) {
        e.variable.name
    }

    def dispatch String toPython(Addition e) '''
(«e.left.toPython» + «e.right.toPython»)
'''

    def dispatch String toPython(Subtraction e) '''
(«e.left.toPython» - «e.right.toPython»)
'''

    def dispatch String toPython(Multiplication e) '''
(«e.left.toPython» * «e.right.toPython»)
'''

    def dispatch String toPython(Division e) '''
(«e.left.toPython» / «e.right.toPython»)
'''
}
